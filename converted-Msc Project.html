	\section{Introduction}
	In Algebraic Geometry we study the relationship between Geometric objects and Algebraic objects.So throughout we will try to construct a dictionary between Algebraic and Geometric objects.And in the end we prove the Equivalence of categories between the Category of Affine Variety over $k$ and Category of finitely generated integral domain over $k$.  \\
	We first define univariate polynomial ring:-
\begin{definition}	Let $R$ be a ring. Then,
	\[R[X]=\{a_0+a_1X_1+a_2X_2^{2}+\cdots +a_nX_n^{n}| \ a_i\in R \ \text{and} \ a_n\neq 0\}\]
\end{definition}
	Similarly we can consider $R[X_1,...,X_n]$ as $R[X_1,...X_{n-1}][X_n]$\\
	i.e. Polynomial ring with indeterminate $X_n$ and coefficient from polynomial in ring $R[X_1,...,X_{n-1}]$
	\begin{definition}\thlabel{Z(S)}
		We define zero set  or zero locus of set of polynomial $S$ in $R[X_1,...,X_n]$ as follows:-\\
		$Z(S)=\{ (a_1,...,a_n)\in R^n \ | \ f(a_1,...,a_n)=0 \ \forall f \in S \}$ \end{definition}
	\begin{example}
	 An affine plane curve is the zero set of one complex polynomial in the 
	complex plane $\mathbb{C}^2$.
	\end{example}
	It can happen that $Z(S)$ is empty.
	\begin{example}
		Let $R=\mathbb{R}$ and consider single equation in one variable $f(x)=x^2+1$ then zero set is $\varnothing $ over $\mathbb{R}$\end{example}
	We don't want $Z(S)$ to be empty.
	And Hilbert's Nullstellenzats  gives us the assurance that Z(S) won't be empty when working over Algebraically closed field.\\
	\textbf{So throughtout we will be working over algebraically closed field $k$}.

	\begin{definition}[Minimal Prime Ideal]
	A prime ideal $\mathfrak{p}$ is said to be minimal prime ideal over an ideal $I$ if it is minimal among all prime ideals containing $I$.(If $I$ is prime then $I$ is the only minimal prime over it).\end{definition}
	\begin{definition}[Krull dimension]
	Krull dimension of ring $R$ is the supremum of the lengths of chains of prime ideals in $R$. Denoted by dim$R$.\end{definition}
	Now, $R$ be a ring and $I$ be prime ideal then codim$I$(also called height of $I$ or rank$I$) is by definition dimension of local ring $R_I$.\\
	If $I$ is not prime then we define codim$I$ to be minimum of the codimensions of primes containing $I$.

	A ring $R$ is called Artinian if it satisfies Descending Chain Condition on ideals.

\begin{theorem}\thlabel{th2.1}
	If $R$ is Artinian integral domain then $R$ is a field.
	\begin{proof}
		Let $x\in R$ be non-zero element.\ Consider\\ $(x)\supseteq (x^2)\supseteq \cdots$ which stabilizes, so there exists $m\in \mathbb{Z^{+}}$ such that for $n\geq m$ $(x^{n})=(x^{n+1})$.\ Therefore $x^n=ax^{n+1}$ for some $a\in R$.\ Since $R$ is integral domain we have $1=ax$ and the result follows.
	\end{proof}
	\end{theorem}
\begin{corollary}
	In Artinian ring every prime ideal is maximal.
\begin{proof} By \thref{th2.1} and the fact that $R/p$ is a field if and only if $p$ is maximal ideal.\end{proof}

\end{corollary}

	\begin{definition}[Symbolic Power]\thlabel{Symbolic Power}
	Let $p$ be prime ideal in ring $R$ and $n\in \mathbb{N}$. The following ideal is called $n^{th}$ symbolic power of $p$.
	\[p^{(n)}=p^nR_p\cap R=\{ a\in \mathbb{R}| \ ab\in p^n \ \text{for some} \ b\in R\setminus p \}\]
\end{definition}
\begin{remark}\thlabel{supset} For prime ideal $p$ we have $p^{(1)}\supseteq p^{(2)}\supseteq p^{(3)}\cdots$ \end{remark}
	\begin{lemma}\thlabel{symbolic 1} $p^n\subseteq p^{(n)}\subseteq p$
	\begin{proof}
	Let $a\in p^n$ then $a\cdot 1\in p^n \implies a\in p^{(n)} \implies p^n\subseteq p^{(n)}$.\\
	$a\in p^{(n)} \implies ab \in p^n (b\in S) \implies ab\in p \implies a \in p$\end{proof} \qed \end{lemma} 
	\begin{lemma} $p^{(n)}$ is $p-$primary.
	\begin{proof}
	Since $p$ is prime, $\sqrt{p}=p$.\ Therefore
	 by above lemma $\sqrt{p^n}=p\subseteq \sqrt{p^{(n)}}\subseteq \sqrt{p}$.It implies $\sqrt{p^{(n)}}=p$.\\
	To show $p^{(n)}$ is primary\\
	let $ab\in p^{(n)}$ or $abx\in p^n$ for $x\in R\setminus p$\\
	i.e. To show $a\in p^{(n)} \ or \ b^m\in p^{(n)}$\\
	If $b\notin \sqrt{p^{(n)}}=p \implies b\in R\setminus p$.\\
	We have $abx\in p^n$ with $bx\in R\setminus p.So,\ a\in p^{(n)}$.\ Hence, $p^{(n)}$ is $p-$ primary.\end{proof} \qed \end{lemma}
	\begin{lemma} $p^{(n)}R_p=p^nR_p$.
    \begin{proof}
	Since $p^n\subseteq p^{(n)}\subseteq p$ it implies that $p^nR_p\subseteq p^{(n)}R_p$\\
	To show: $p^{(n)}R_p\subseteq p^nR_p$\\
	Let $\frac{b}{s}\in p^{(n)}R_p$.
	Now $b\in p^{(n)}$ implies $bc\in p^n$ for some $c\in R\setminus p$.\ Therefore,
	 $\frac{b}{s}=\frac{bc}{sc}\in p^nR_p$ and 
	 $p^{(n)}R_p\subseteq p^nR_p$.
	Thus,\\ $p^{(n)}R_p=p^nR_p$. \end{proof} \qed \end{lemma}

\end{proof}
\end{theorem} 
	\begin{theorem}\thlabel{KPID}[Krull's Principal ideal theorem]
		Let R be\\ Noetherian ring.If $x\in R$ be non-zero and non-unit and $P$ is minimal among prime ideals containing $(x)$, then codim$(P)\leq 1.$
	\begin{proof}
	Let $x\in R$ and $P$ be the minimal prime ideal containing $(x)$. We show that if $Q\subsetneq P$ is prime then dim$R_Q=0$ and thus codim$(P)\leq 0$.\\
	From \thref{Correspondence theorem} we have Codim$(P)=$dim$(R_P)$. Let $S=R\setminus P$ and $x\in P$ then $PR_P$ contains $\frac{x}{1}$.Suppose there is another prime ideal in $R_P$ containing $\frac{x}{1}$ say $qR_P$ with $q\subset P$ then $\frac{x}{1} \in qR_P$ and it will imply $x\in q$ which is contradiction since $P$ is the minimal prime containing $(x)$ and is unique prime ideal of $R_P$.Hence we have $R_P$ is Noetherian, local ring with $PR_P$ minimal prime ideal over $x$ such that codim$(PR_P)=$codim$(P)$.Therefore without loss of generality assume $R$ is local ring with maximal ideal $P$.\\
	Now since $P$ is minimal over $(x)$.Prime ideals of $R/(x)$ are prime ideals of $R$ containing $(x)$. i.e.\ in this case $P$. So dim$R/(x)=0$ and $R/(x)$ is Noetherian and by \thref{Artin ring} $R/(x)$ is Artinian.\\
	Thus from above and \thref{supset} the chain \[(x)+Q\supseteq (x)+Q^{(2)} \supseteq Q^{(3)} \cdots\] stabilizes so there exists $n$ such that $(x)+Q^{(n)}=(x)+Q^{(n+1)}$\\

	By \thref{Symbolic Power} $axb\in Q^{(n)}$ for $b\in R\setminus Q$. Also, $x\notin Q$ implies that $xb\in R\setminus Q$ and $a\in Q^{(n)}$. It implies \\ $f\in Q^{(n)}(x)+Q^{(n+1)}$ and $Q^{(n)}\subseteq Q^{(n)}(x)+Q^{(n+1)}$.Also, by \thref{supset} $Q^{(n)}\supseteq Q^{(n+1)}$ implies $Q^{(n)}\supseteq Q^{(n)}(x)+Q^{(n+1)}$ and hence our claim follows.Now,quotienting out both sides by $Q^{(n+1)}$ we get $Q^{(n)}/Q^{(n+1)}=(x)Q^{(n)}/Q^{(n+1)}$.Therefore by \thref{Nakayama} \\ $Q^{(n)}/Q^{(n+1)}=0$ and it implies $Q^{(n)}=Q^{(n+1)}$. \\

	\end{proof}

		\begin{proof} We have
			$\displaystyle\prod_{k=1}^m(S_k)=\{f_1\cdot \cdot \cdot f_m | f_i\in S_i\} $\\
			Let $(t_1,...,t_n)\in \displaystyle\bigcup_{k=1}^mZ(S_k)$ then $(t_1,...,t_n)\in Z(S_i)$ for atleast one i.\\ It implies $(t_1,..,t_n) \in Z(S_1 \cdot \cdot \cdot S_m).$ Since the product will vanish even if one polynomial of product vanishes.Hence,
			
				\[\displaystyle\bigcup_{k=1}^mZ(S_k) \subseteq Z(\prod_{k=1}^m(S_k))\]

			Suppose $(t_1,...,t_n) \notin Z(S_i)$ for every i\\
			then $\forall i$ $\exists$ $g_i \in S_i$ such that $g_i(t_1,...,t_n) \neq 0.$
			And we have \\ $g_1 \cdot \cdot \cdot g_m \in S_1 \cdot \cdot \cdot S_m$ but $(g_1 \cdot \cdot \cdot g_m)((t_1,...,t_n)) \neq 0$ which is a contradiction to our assumption.\\ So,
				\[Z(\prod_{k=1}^m(S_k)) \subseteq \displaystyle\bigcup_{k=1}^m Z(S_k)\] \end{proof}
	Hence the lemma $\displaystyle\bigcup_{k=1}^mZ(S_k)=Z(\prod_{k=1}^m)$\end{lemma}

	\begin{lemma}$\displaystyle \bigcap_{m}Z(S_m)=Z(\bigcup_{m} S_m)$

Hence the lemma	$ \displaystyle \bigcap_{m}Z(S_m)=Z(\bigcup_{m} S_m)$\end{lemma}

	\begin{definition}[Cateorgy]
		A category $\mathbf{C}$ can be described as a set $\mathbf{Ob}$, whose members are the objects of $\mathbf{C}$, satisfying the following three conditions:\\
		$Morphism$\ :For every pair $X,Y$ of objects, there is a $\mathbf{Hom}(X,Y)$ called the $morphisms$ from $X$ to $Y$ in $\mathbf{C}$.If $f$ is a morphism from $X$ to $Y$, then we denote $f:X\longrightarrow Y$.\\
		$Identity$\ : For every object $X$, there exists a morphism id$_X$ in $\mathbf{Hom}(X,X)$ called identity on $X$.\\
		$Composition$\ : For every triple $X,Y,Z$ of objects, there exists a partial binary operation from
		$\mathbf{Hom}(X,Y)\times \mathbf{Hom}(Y,Z)$ to $\mathbf{Hom}(X,Z)$, called the composition of morphisms in $\mathbf{C}$. If $f:X\longrightarrow Y$ and $g:Y\longrightarrow Z$ then the composition of $f$ and $g$ is denoted $g\circ f:X\longrightarrow Z$.\\
		Further Identity,\ composition and morphisms satify the two axioms:\\
		$Asscociativity$\ : If $f:X\longrightarrow Y$,\ $g:Y\longrightarrow Z$ and $h:Z\longrightarrow W$,\ then $h\circ (g\circ f)=(h\circ g)\circ f$.\\
		$Identity$\ : If $X\longrightarrow Y$ then (id$_Y\circ f)=f$ and $(f \circ$ id$_X)=f$.
		\end{definition}
	\begin{example}
		Consider the category $Vec_{k}$ of vector spaces over the given field $k$.Here the objects are $k$-vector spaces and the morphisms are linear trabsformations.
	\end{example}
	\begin{example}
		Category of abelian groups $Ab$, consists of abelian groups as objects and group homomorphisms as the morphisms.
	\end{example}
	\begin{definition}[Contravariant Functor]
		A contravariant functor $F$ from  a category $\mathcal{A}$ to a category $\mathcal{B}$, denoted $F:\mathcal{A}\longrightarrow \mathcal{B}$ is the following data. It is a map of objects $F:\text{obj}(\mathcal{A})\longrightarrow \text{obj}(\mathcal{B})$, and for each $A_1,A_2\in \mathcal{A}$,\ and for morphism $m:A_1\longrightarrow A_2$,\ a morphism $F(m):F(A_2)\longrightarrow F(A_2)$ in $\mathcal{B}$.\ We further require that $F$ preserves identity morphism. i.e.\ for $A\in \mathcal{A}$,\ $F(\text{id}_A)=\text{id}_{F(A)}$,\ and that $F$ preserves composition. i.e.\ $F(m_2\circ m_1)=F(m_1)\circ F(m_2)$.
		\end{definition}
	\begin{example}
		$Vec_{k}$ is the category of $k$ vector spaces, then dual gives contravariant functor $(.)^*:Vec_{k}\longrightarrow Vec_{k}$.\ For each linear transformation between vector spaces,\  $f:V\longrightarrow W$ we have dual transformation
		$f^*:W^*\longrightarrow V^*$ and $(f\circ g)^*=g^*\circ f^*$.
	\end{example}
	\begin{example}
		There is a contravariant functor $Top\longrightarrow Rings$ taking a topological space $X$ to the ring of real-valued continuous functions on $X$.\ A morphism
		of topologiacal spaces $X\longrightarrow Y$,\ induces the pullback map from functions on $Y$ to functions on $X$. 
	\end{example}
\begin{definition}[Equivalence of Categories]
	Given two categories $\mathcal{A}$ and $\mathcal{B}$,\ equivalence of categories consists of functors $F:\mathcal{A}\longrightarrow \mathcal{B}$ and $F':\mathcal{B}\longrightarrow \mathcal{A}$ such that $F\circ F'$ is natuarlly isomorphic to the identity functor $\text{id}_{\mathcal{B}}$ on $\mathcal{B}$ and $F'\circ F$ is naturally isomorphic to $\text{id}_{\mathcal{A}}$.
\end{definition}
In \ref{Morphism} we will use alternative characterization of equivalence of categories:\\
A functor $F:\mathcal{A}\longrightarrow \mathcal{B}$ yields equivalence of categories if and only if it simultaneously:\\
$a$) $Full$ i.e.\ for any two objects $A_1,A_2\in \mathcal{A}$, the map\\ $\mathbf{Hom}_{\mathcal{A}}(A_1,A_2)\longrightarrow \mathbf{Hom}_{\mathcal{B}}(F(A_1),F(A_2))$ induced by $F$ is\\ surjective.\\
$b$) $Faithful$ i.e.\ for any two objects $A_1,A_2\in \mathcal{A}$, the map\\ $\mathbf{Hom}_{\mathcal{A}}(A_1,A_2)\longrightarrow \mathbf{Hom}_{\mathcal{B}}(F(A_1),F(A_2))$ induced by $F$ is\\ injective.\\
$c$) $Essentially$ $surjective(dense)$ i.e.\ each object $B_1\in \mathcal{B}$ is \\ isomorphic to an object of the form $F(A_1)$,\ for $A_1 \in \mathcal{A}$.

\end{remark}

	\subsection{Regular functions}
	\begin{definition}\thlabel{regular function} Let $f$ be a function defined in an open neighborhood of a point $x$ in an affine variety or quasi affine variety then $f$ is regular at $x$ if there exists polynomial $g$ and $h$ with $h\neq 0$ in an open neighborhood of $x$ and $f=\frac{g}{h}$ in that open neighborhood of $x$ and $f$ is regular function if it regular at each point of space.
	\end{definition}
	\begin{flushleft}Let $U\subset \mathbb{A}_k^n$ be affine variety or quasi affine variety then \\ $\mathcal{O}(U)=\{f| f$ is regular on $U\}$.
		$f\in \mathcal{O}(U) \implies$ $f:U \Longrightarrow k$ is a map that is regular at each $x \in U$.So for every $x\in U$ there exists polynomial $g_x$,$h_x\in k[X_1,\cdots ,X_n]$ such that $x\in D(h_x)$ and $f=\frac{g_x}{h_x}$ on an open neighborhood of $x$ contained in $D(h_x)$.
	\end{flushleft}
\newpage
	\begin{lemma}
		Any regular function is continuous.
		\begin{proof}
			Let $f\in \mathcal{O}(U)$ and $f=\frac{g}{h}$ for some polynomials $g$ and $h$.It is enough to prove $f^{-1}$ of closed set is closed.A closed set of $\mathbb{A}_k^1$ is finite set of points since degree of any polynomial is taken to be finite.Also $f^{-1}(\bigcup\limits_{i} A_i)=\bigcup\limits_{i}(A_i)$ so it is enough to show that $f^{-1}(a)=\{p\in Y|f(p)=a\}$ is closed for any $a\in k$.
			A subset $Z$ of $Y$ is closed if and only if $Y$ can be covered by open subsets $U$ such that $Z\cap U$ is closed in $U$ for every $U$.Let $U$ be the open set on which $f$ can be represented as $f=\frac{g}{h}$ with $g,h\in A(\mathbb{A}_k^n)$ and $h$ is nowhere $0$ on $U$.Then, \[f^{-1}\cap \ U=\biggl\{p\in U|\frac{g(p)}{h(p)}=0 \biggr\}=\{p\in U| p\in Z(g-ah)\}= U\cap Z(g-ah)\]So $f^{-1}(a)$ is closed in $Y$ and the result follows. 
		\end{proof}
	\end{lemma}
	\begin{lemma} If two regular functions are equal on non-empty open subset of irreducible topological space $Y$ then they are equal everywhere.
		\begin{proof} Any non-empty open subset is dense.Let $f,g\in \mathcal{O}(A)$ and $f=g$ on some $U\subseteq Y$. Then on $U$,$f-g=0 \implies U\subset Z(f-g)$ and $U$ is dense so $Z(f-g)=Y$  and the result follows that $f=g$.
		\end{proof}
	\end{lemma}
	\begin{theorem}\thlabel{exact} $\mathcal{O}(\mathbb{A}_k^n) \simeq A(\mathbb{A}_k^n)$
		\begin{proof} Define the map as follows
			\begin{align*}
				\varphi:A(\mathbb{A}_k^n) &
				\longrightarrow \mathcal{O}     (\mathbb{A}_k^n) \\
				g &\longmapsto (g:\mathbb{A} _k^n \longrightarrow k)
			\end{align*}
			Every polynomial is regular function and $\varphi$ is $k$-algebra homomorphism.Since $k$ is algebraically closed field $ker\varphi=\{0\}$.So $\varphi$ is injective.\\
			Let $f:\mathbb{A}_k^n\longrightarrow k$ be a mapping.$\mathbb{A}_k^n$ is open set so there exists finitely many points $x_1,x_2,\cdots ,x_n$ such that $\mathbb{A}_k^n=\bigcup\limits_{i=1}^{m}D(h_{x_i})$ where $h_{x_i}\in k[X_1,\cdots ,X_n]$ and $f=\frac{g_{x_i}}{h_{x_i}}$.So we have,\\ $\emptyset=(\bigcup\limits_{i=1}^{m}D(h_{x_i}))^{c}=\bigcap\limits_{i=1}^{m}Z(h_{x_i})=Z(h_{x_i},\cdots ,h_{x_m})$.It implies \\ $(h_{x_i},\cdots ,h_{x_m})=1$.Then there exists $t_1,\cdots ,t_m\in k[X_1,\cdots ,X_n]$ such that $\sum\limits_{i=1}^{m}t_ih_{x_i}=1$.Therefore, $\sum\limits_{i=1}^{m}(t_ih_{x_i})f=f$ and $\sum\limits_{i=1}^{m}t_ig_{x_i}=f.$
		\end{proof}
	\end{theorem}
	\subsection{Morphisms of affine and quasi affine varieties} \label{Morphism}
	In this section by varieties we mean affine or quasi affine varieties.
	\begin{definition}\thlabel{morphism} Let $X$ and $Y$ be varieties.A morphism $\varphi$ from $X$ to $Y$ is continuous map.Further,$\varphi:X\longrightarrow Y$ is such that for all open subsets $V\subset Y$, the pullback of maps via $\varphi$ takes regular function on $V$,namely $\mathcal{O}(V)$ to regular functions on $U$,namely $\mathcal{O}(U)$ where $U=\varphi^{-1}(V)$ is open.
	\end{definition}
	\begin{theorem} Mor$_{var}(X,\mathbb{A}_k^1)=\mathcal{O}(X)$
	\begin{proof}Let $\varphi:X\longrightarrow \mathbb{A}_k^1$ be a morphism.So for any open set $V\subset \mathbb{A}_1^k$ we have $\varphi^{*}:\mathcal{O}(V)\longrightarrow \mathcal{O}(\varphi^{-1}(V))$.Put $V=\mathbb {A}_k^1$.\ Due to \thref{exact} we have $\mathcal{O}(\mathbb{A}_k^1)=A(\mathbb{A}_k^1)=k[X]$.\ So and $\varphi\in \mathcal{O}(X)$.
		\begin{align*} \varphi^{*}:\mathcal{O}(\mathbb{A}_k^1)&\longrightarrow \mathcal{O}(X) \\
		 \text{id}_{\mathbb{A}_k^1}&\longmapsto \varphi^{*}(\text{id}_{\mathbb{A}_k^1})=\text{id}_{\mathbb{A}_k^1}\circ \varphi=\varphi
	\end{align*}
   \
	Conversely,\ let $f\in \mathcal{O}(X)$.Then by \thref{morphism} $f$ is continuous.We check $\forall$ open subsets $V\subset \mathbb{A}_k^1$ and every $\psi\in \mathcal{O}(V)$,\ $f^{*}(\psi)=\psi\circ f\in \mathcal{O}(f^{-1}(V))$.\\
	\underline{claim}: Non-empty open subset of $\mathbb{A}_k^1$ is basic affine open.\\
	Let $V\subseteq \mathbb{A}_k^1$ be open.Then $V^{c}=\{\lambda_1,\cdots ,\lambda_n \}$ where $\lambda_i\in \mathbb{A}_k^1$. Since $k[X]$ is principal ideal domain, for $g\in k[X]$,\\ $Z(g)=\{\lambda_1,\cdots ,\lambda_n \}=V^{c}$.Thus $V=D(g)$.\ Consider the following diagram
    \begin{center}
	\begin{tikzcd}
		X \arrow[rr, "f"]                                          &   & \mathbb{A_k^1}                                              \\
		f^{-1}(V) \arrow[u, "open", hook] \arrow[rd, "f^{*}(\frac{h}{g^m})"'] \arrow[rr, "f"] &   & V=D(g) \arrow[u, "open"', hook] \arrow[ld, "\frac{h}{g^m}"] \\
		& k &                                                
	\end{tikzcd} \end{center}
\underline{claim}:$f^{*}(\frac{h}{g^m})=\frac{h\circ f}{(g\circ f)^m}$ is regular function.\\
$g\circ f$ does not vanish on $f^{-1}(V)$ because if $g\circ f$ vanishes on $f^{-1}(V)$ then $(g\circ f)(x)=0$ for $x\in f^{-1}(V) \implies g(f(x))=0$ but $f(x)\in D(g)$	where $g$ does not vanish.So it is enough to show that $h\circ f$ and $g\circ f$ are regular function on $f^{-1}(V)$. $f$ is regular so locally $f=\frac{P}{Q}$ where $P,Q\in k[X_1,\cdots ,X_n]$ with $X \hookrightarrow \mathbb{A}_k^n$ and if $h=h(t)=\sum\limits_{i=1}^{m}a_it^i$ then $h\circ f$ is locally $h(\frac{P}{Q})=\sum\limits_{i=1}^{m}\frac{a_iP^i}{Q}$.\ So $h\circ f$ is regular on $f^{-1}(V)$ and $f\in$Mor$_{var}(X,\mathbb{A}_k^1)$.
\end{proof}
\end{theorem}

\begin{lemma}\thlabel{ZLRF} Let $U$ be an open subset of an affine variety $X$ and let $\varphi\in \mathcal{O}(U)$.\ Then $V(\varphi)=\{x\in U|\varphi(x)=0\}$ is closed in $U$.
\begin{proof} By \thref{regular function} any point $x\in U$ has an open neighborhood $U_x$ in $U$ on which $\varphi=\frac{g_x}{f_x}$ for some $f_x,g_x\in A(X)$ and $U_x=D(f_x)$. So the set,\\
\begin{align*}
U_x\textbackslash Z(\varphi)=U_x\textbackslash Z(g_x)&=\{y\in U_x| \ \varphi(y)\neq 0\} \\
&=\{y\in U_x| \ g_x(y)\neq 0\}\\
&=\{y\in U_x| \ y\in D(g_x)\}\\
&=U_x\cap D(g_x) \ \text{which is open in} \ X
\end{align*}
So, $\bigcup\limits_{x\in U}U_x\cap D(g_x)$ is open in $X$. Thus $\bigcup\limits_{x\in U} U_x\textbackslash Z(\varphi)=U\textbackslash Z(\varphi)$ is open in $X$.\ This implies $Z(\varphi)$ is closed in $U$ since $U\textbackslash Z(\varphi)\subset U.$
\end{proof}
\end{lemma}
\begin{theorem}\thlabel{main}
Let $X$ be an affine variety and let $Y$ be any variety. Then there is natural bijection of sets.
\begin{align*}
	\text{Mor}_{var}(Y,X) \longleftrightarrow \text{Hom}_{k-algebra}(A(X),\mathcal{O}(Y))
\end{align*}
\begin{proof} Define $\alpha$ as 
	\begin{align*}
		\alpha:\text{Mor}_{var}(Y,X) &\longleftrightarrow \text{Hom}_{k-algebra}(A(X),\mathcal{O}(Y)) \\
		\varphi&\longmapsto \varphi^{*}
	\end{align*}
Also,\ $\varphi^{*}$ is $k-algebra$ homomorphism since addition, multiplication are defined pointwise.\ We show bijection by proving that $\alpha$ is injective and surjective.\\
Suppose $\alpha(\varphi_1)=\alpha(\varphi_2)$ then we need to show that $\varphi_1=\varphi_2$.\ Let $\mathbb{A}_k^n$ be affine space with coordinates $T_1,\cdots ,T_n$ and\\ $\varphi_1,\varphi_2\in \text{Mor}_{var}(Y,X)$.\\
For $y\in Y$ let $\varphi_1(y)=(\lambda_1,\cdots ,\lambda_n)$ and $\varphi_2(y)=(\mu_1,\cdots ,\mu_n)$.Further,by our suppostion we have
\begin{align*}
	\varphi_1^{*}=\varphi_2^{*}:A(X)=\mathcal{O}(X)&\longrightarrow \mathcal{O}(Y)\\
	T_i|_X=\overline{T_i}&\longmapsto \varphi_1^{*}(\overline{T_i})=\varphi_2^{*}(\overline{T_i}) \ \forall i\\ 
\end{align*}
i.e.\ $\overline{T_i}\circ \varphi_1=\overline{T_i}\circ \varphi_2 \ \ \forall i$ $\implies$ $\lambda_i=\mu_i$ \ $\forall i$.\ Hence $\varphi_1(y)=\varphi_2(y)$.\\
Conversely, let	$\zeta\in$ Hom$_{k-algebra}(A(X),\mathcal{O}(Y))$ so\\
\[\zeta:A(X)\longrightarrow \mathcal{O}(Y), \overline{T_i}\longmapsto f_i\]
Define, \[\psi:Y\longrightarrow \mathbb{A}_k^n,\ y\mapsto (f_1(y),\cdots ,f_n(y) )\]
\underline{claim:} $\psi$ maps into $X$.\\
Let $g\in$ I$(X)\subset k[T_1,\cdots ,T_n]$.Then $\bar{g}=0$ in $A(X)=k[T_1,\cdots ,T_n]/$I$(X)$.\\
We have,\[\zeta:A(X)\longrightarrow \mathcal{O}(Y)\] is $k-algebra$ ring homomorphism and hence $\zeta(\bar{g})=0$.\\
i.e.\ $\zeta(g(\overline{T_1},\cdots ,\overline{T_n}))=0\implies g(f_1,\cdots ,f_n)=0$.\ $g(f_1,\cdots ,f_n)$ is polynomial $f_i\in \mathcal{O}(Y) and g(f_1,\cdots ,f_n)=0$. Hence $g(f_1,\cdots ,f_n)$ is zero function.It implies $g(f_1(y),\cdots ,f_n(y))=0$ $\forall y\in Y$. Hence $g(\psi(y))=0$ for arbitrary $g\in$I$(x)$ and $\psi(y)\in Z($I$(X))$.\\
\underline{To show:} $\psi$ is morphism.\\
That is we show $\psi$ is continuous and pullback regular function to regular function. We have,\[\psi:Y\longrightarrow X\] is set theoretic map and
\begin{align*}
	\alpha(\psi)=\psi^{*}:A(X)&\longrightarrow Map(Y,k)\\
	h&\longmapsto \psi^{*}=h\circ \psi
\end{align*}
So, ($\psi^{*}(h))(y)=(h\circ \psi)(y)=h(\psi(y))=h(f_1(y),\cdots ,f_n(y))$.And for	$y\in Y$,\  $h(f_1(y),\cdots ,f_n(y)) =\zeta(h(\overline{T_1},\cdots ,\overline{T_n}))(y)$ so it implies $\psi^{*}=\zeta$.Further we have $\zeta:A(X)\longrightarrow\mathcal{O}(Y)$.\ Therefore $\psi$ pullsback regular function to regular function.\ Let $Z(J)\subseteq X$ then,\\

\begin{align*}
 \phi^{-1}(Z(J))=\psi^{-1}(\bigcap\limits_{h\in J}h^{-1}(0))
 &=\bigcap\limits_{h\in J}\psi^{-1}(h^{-1}(0))\\
 &=\bigcap\limits_{h\in J}Z(h\circ \psi) 
\end{align*}
 
\end{minipage}

	\begin{tikzcd}
		Y \arrow[rr, "\psi"] \arrow[rrdd, "\psi^{*}(h)"'] &  & X \arrow[dd, "h"] \\
		&  &                   \\

which is closed in $Y$ by \thref{ZLRF}. \ Hence, $f$ is continuous and $\alpha$ is surjective.
\end{proof}
\end{theorem} 
\begin{lemma}\thlabel{composition} 
If	\begin{tikzcd}
		Y \arrow[r, "\varphi"] & X_1 \arrow[r, "\psi"] & X_2
	\end{tikzcd}  is a sequence of morphism of \\ varieties with $X_1$ and $X_2$ affine, \ then $k-algebra \ homomorphism$ $A(X):\longrightarrow \mathcal{O}(Y)$ given by $(\psi\circ \varphi)^{*}$ is same as composition $\varphi^{*}\circ \psi^{*}$.
\begin{proof} If \begin{tikzcd}
		Y \arrow[r, "\varphi"] & X_1 \arrow[r, "\psi"] & X_2
	\end{tikzcd} then
\begin{tikzcd}
	Y \arrow[r, "\psi\circ \varphi"] & X_2 \arrow[r, "h"] & k
\end{tikzcd}\\
To show that $(\psi\circ \varphi)^{*}=\varphi^{*}\circ \psi^{*}$.
We have,\\
\begin{center}
\begin{tikzcd}
	Y \arrow[r, "\varphi"] \arrow[rrd, "\varphi^{*}(\psi^{*}(h))"', bend right] & X_1 \arrow[r, "\psi"] \arrow[rd, "\psi^{*}(h)"'] & X_2 \arrow[d, "h"] \\
	&                                                  & k                 
\end{tikzcd}\\
\end{center}
$(\psi\circ \varphi)^{*}(h)=h\circ (\psi\circ \varphi)=(h\circ \psi)\circ \varphi$ and\\
$(\varphi^{*}\circ \psi^{*})(h)=\varphi^{*}\circ (\psi^{*}(h))=\varphi^{*}(h\circ \psi)=(h\circ \psi)\circ \varphi$
\end{proof}
\end{lemma}

\begin{proposition} Two Affine Varieties $X_1$ and $X_2$ are isomorphic if and only if $A(X_1)$ and $A(X_2)$ are isomorrphic as $k-algebras$.
\begin{proof} From \thref{main} 
	\begin{align*}
		\alpha:\text{Mor}_{var}(X_1,X_2) &\xlongrightarrow{\sim} \text{Hom}_{k-algebra}(A(X_2),A(X_1))\\
		\varphi&\longmapsto \alpha(\varphi)=\varphi^{*}
	\end{align*}
If $X_1$ and $X_2$ are isomorphic and $\varphi\in$Mor$_{var}(X_1,X_2)$ the there exists $\psi=\varphi^{-1}$ such that $\varphi\circ \psi=$id$_{X_2}$ then since composition is a morphism and from \thref{composition} we have,\[ \text{id}_{X_2}=\varphi\circ \psi \longmapsto (\varphi\circ \psi)^{*}=\psi^{*}\circ \varphi^{*}=\text{id}^*\]
This proves $\varphi^*$ is an isomorphism.\ Similar argument hold for the converse part.
\end{proof}
\end{proposition}
